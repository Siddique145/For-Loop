// 2. Summation with For Loop: console.log("\n2. Summation with For Loop:"); let sum = 0; for (let i = 1; i <= 10; i++) { sum += i; } console.log("Sum of numbers from 1 to 10:", sum);
// 3. Reverse Counting with For Loop: console.log("\n3. Reverse Counting with For Loop:"); for (let i = 10; i >= 1; i--) { console.log(i); }
// 4. Multiplication Table with For Loop: console.log("\n4. Multiplication Table with For Loop:"); const number = 5; // You can change this number to generate multiplication table for different numbers for (let i = 1; i <= 10; i++) { console.log(${number} * ${i} = ${number * i}); }
// 5. Conclusion: console.log("\n5. Conclusion:"); console.log("For loops in JavaScript provide a convenient way to iterate over a range of values or elements in an array. They are fundamental control flow structures in programming, allowing for repetitive execution of code based on specified conditions.");
// 6. Pattern Printing with For Loop: console.log("\n6. Pattern Printing with For Loop:"); for (let i = 1; i <= 5; i++) { let row = ''; for (let j = 1; j <= i; j++) { row += '* '; } console.log(row); }
